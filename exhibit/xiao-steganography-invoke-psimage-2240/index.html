.SYNOPSIS
    Embeds a PowerShell script into the pixels of a PNG image.
.DESCRIPTION
    Hides a powershell script within a PNG image using steganography.  The image is written to disk and can be executed with:(...omitted for brevity, standard PowerShell script header comments...)

function Invoke-PSImage
{
    [CmdletBinding(DefaultParameterSetName="Default")]
    param(
        [Parameter(Position=0,Mandatory=$true,ParameterSetName="Default")]
        [ValidateScript({Test-Path $_})]
        [string]
        $Image,

        [Parameter(Position=1,Mandatory=$true,ParameterSetName="Default")]
        [ValidateScript({Test-Path $_})]
        [string]
        $Script,

        [Parameter(Position=2,Mandatory=$true,ParameterSetName="Default")]
        [string]
        $Out,
		
        [Parameter(ParameterSetName="Sandbox")]
        [switch]
        $Sandbox
    )
    
    Write-Verbose "[-] Starting Invoke-PSImage"

    # (...omitted for brevity...  Variable declarations and setup...)

    #Read Image
    try{
       $img = [System.Drawing.Image]::FromFile((Get-Item $Image).FullName)
    }
    catch [System.OutOfMemoryException] {
	   Write-Verbose $_
       Write-Warning  "[-] It appears as though the input image file is not a valid image.  Did you use a .jpg and just rename to .png?"
       Write-Warning  "[-] If it's valid, perhaps the image is too large for memory. Try running this process x64 bit powershell."
	   exit 1
    }
    catch{
        Write-Warning "[-] Error reading image.  Make sure you have the correct path, and the file is a VALID image."
		Write-Verbose $_
        Exit 1
    }
    $width = $img.Size.Width
    $height = $img.Size.Height
    Write-Verbose ("[*] Reading image data from {0}" -f (Get-Item $Image).Name)   
    $bitmap = New-Object System.Drawing.Bitmap $width,$height
    $bitmap = [System.Drawing.Bitmap]::FromFile((Get-Item $Image).FullName)
    $img_stream = New-Object System.IO.MemoryStream
    $img.save($img_stream, [System.Drawing.Imaging.ImageFormat]::Png)
    $image_bytes = $img_stream.ToArray()
    $img_stream.Dispose()
    $image_size = $image_bytes.Count

        
    #----------------------------------------
    #Read and Encode the Payload Script
    #----------------------------------------
    # (...omitted for brevity...  Reading and encoding the script file content...)
    try{
		$fileToEncode = Get-Content $Script
    }
    catch{
        Write-Warning "[-] Please ensure that the file exists, and that the correct path was specified."
        Exit 1
    }
    $encScript = [System.Text.Encoding]::Unicode.GetBytes($fileToEncode)
    [byte[]] $encScriptPad = @(0,0,0,0,0,0,0,0,0) #9 bytes to pad for the 3 pixel header.
    Write-Verbose "[+] Script payload accepted."
    [byte[]] $encScript = $encScriptPad + $encScript

    #(...omitted for brevity...  Functions ConvertTo-Pixel and Get-PixelColor are defined... These functions convert byte data to RGB pixel colors and vice versa...)
    function ConvertTo-Pixel
    {
      param([byte[]]$bytes)
      $pixel = @()
      $bytes | % {
        $pixel += $_
        }

       if($pixel.Count -eq 3){
        return ([System.Drawing.Color]::FromArgb($pixel[0],$pixel[1],$pixel[2]))
       }

       if($pixel.Count -eq 2){
        return ([System.Drawing.Color]::FromArgb($pixel[0],$pixel[1],0))
       }

       if($pixel.Count -eq 1){
         return ([System.Drawing.Color]::FromArgb($pixel[0],0,0))
       }
    }

    function Get-PixelColor
    {
        param([System.Drawing.Color]$color)
        return $color.R, $color.G, $color.B
    }

    #-------------------------------
    #Build the script stream
    #-------------------------------
    # (...omitted for brevity... Logic to calculate required pixels and resize if necessary...)

    [System.Collections.ArrayList]$script_stream = @()
    $encScript | % {
        [void]$script_stream.Add($_)
    }

    #--------------------------------
    #Build the output image
    #--------------------------------
    Write-Verbose "[*] Building output file..."
    $x = 0
    $y = 0
    $encLen = $encScript.Length

    0..($encLen / 3) | %{
        $pixelIndex = $_ * 3
        $colors = Get-PixelColor -color (ConvertTo-Pixel -bytes $script_stream[$pixelIndex..($pixelIndex + 2)])
        $bitmap.SetPixel($x,$y, ([System.Drawing.Color]::FromArgb($colors[0],$colors[1],$colors[2])))
        if($x -lt $width - 1){
            $x++
        }

The script reads an input PNG image ($Image) and a PowerShell script ($Script). It encodes the PowerShell script into Unicode bytes ($encScript). It then iterates through the bytes of the encoded script, converting groups of three bytes into RGB color values. These RGB values are then used to set the pixel colors of a new bitmap, effectively embedding the script data within the image's pixel data. Finally, the modified bitmap is saved as a new PNG image ($Out). The key functions are ConvertTo-Pixel (takes bytes, returns a Color) and Get-PixelColor (takes a Color, returns bytes). The script modifies the pixels sequentially, wrapping around to the next row when it reaches the end of a row. Page 46 through 50 contain the PowerShell script.
