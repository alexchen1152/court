      #!/usr/bin/python
import os
import sys
import time
import subprocess


####################
# Main script to run all of the SQL statements found in this directory.
#       SQL statements are located in the current folder named *.sql and
#      are executed in name order.
#
#      Each sql statement output is saved into two (2) log files:
#               sqlout[sqlfilename]_out.txt - Contains standard output
#               sqlout[sqlfilename]_err.txt - Contains error output
#
#     The SQL*PLUS output is also redirected to nul for standard output for
#     cleanliness.  Error output, if any, will be displayed on the screen
#     for troubleshooting.
#
####################

#Variable statements.
curdir = os.getcwd()
idfile = "allids.txt"
numprocs = 0
maxprocs = 25

#Function to execute sql statements.
def run_sql(stmt,sqlout_fileout,sqlout_fileerr):
        sqlplus = subprocess.Popen(["sqlplus","-s","user/password@database"], stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        sqlplus.stdin.write("WHENEVER SQLERROR EXIT SQL.SQLCODE;\n")
        sqlplus.stdin.write("set linesize 2000;\n")
        sqlplus.stdin.write("set pagesize 2000;\n")
        sqlplus.stdin.write("set serveroutput on;\n")
        sqlplus.stdin.write("@"+stmt+";\n")
        sqlplus.stdin.write("exit;\n")
        (sout, serr) = sqlplus.communicate()

        # Check for any standard output to file.
        if sout is not None :
                outfile = open(sqlout_fileout, 'wb')
                outfile.write(sout)
                outfile.close()

        # Check for any error output to file.
        if serr is not None :
                errfile = open(sqlout_fileerr, 'wb')
                errfile.write(serr)
                errfile.close()
        return;

# Make sure that the allids.txt file exists.
if not os.path.isfile(idfile):
    print "ERROR: All IDs file ("+idfile+") could not be found, please run importids.sh first."
    sys.exit(1)

#Get all the statements and loop through them to submit multi-threaded.
allstmts = [f for f in os.listdir('.') if os.path.isfile(f) and f.endswith('.sql')]
allstmts.sort()

#Loop over statements and submit.
for stmt in allstmts:

    # Get current count of processes running.
    while(numprocs >= maxprocs):
        time.sleep(2)
        numprocs = subprocess.check_output("ps -ef | grep \"sqlplus\" | grep -v grep | wc -l", shell=True)
        numprocs = int(numprocs)
        print str(numprocs)+" current sqlplus processes"

    #Get the current count of processes running.
#    print "Preparing SQL statement: "+stmt
    numprocs = subprocess.check_output("ps -ef | grep \"sqlplus\" | grep -v grep | wc -l", shell=True)
    numprocs = int(numprocs)
    print str(numprocs)+" current sqlplus processes"

    #Submit the job.
    print "  Submitting SQL statement: "+stmt
    sqlout_fileout = "sqlout"+stmt.split(".")[0]+"_out.txt"
    sqlout_fileerr = "sqlout"+stmt.split(".")[0]+"_err.txt"
    run_sql(stmt, sqlout_fileout, sqlout_fileerr)

#Wait for processes to complete
print "Waiting for all processes to complete...."
while(numprocs > 0):
    time.sleep(5)
    numprocs = subprocess.check_output("ps -ef | grep \"sqlplus\" | grep -v grep | wc -l", shell=True)
